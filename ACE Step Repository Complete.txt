import subprocess
import time
import sys
import os

print("⏳ Starting API Server with Offloading (for ~15GB GPU)...")

# Define Environment Variables for Offloading
env = os.environ.copy()
env["ACESTEP_OFFLOAD_TO_CPU"] = "1"      # Critical for T4
env["ACESTEP_OFFLOAD_DIT_TO_CPU"] = "1"  # Optional, but safer
env["PYTORCH_CUDA_ALLOC_CONF"] = "expandable_segments:True" # Help fragmentation

# 1. Start the server (background)
log_file = open("server.log", "w")
process = subprocess.Popen(
    ["nohup", "uv", "run", "./run_server.sh"],
    stdout=log_file,
    stderr=subprocess.STDOUT,
    start_new_session=True,
    env=env
)

print("🚀 Server started with CPU Offload! Monitoring logs...")

# 2. Monitor detailed progress
start_time = time.time()
server_ready = False

while True:
    try:
        with open("server.log", "r") as f:
            lines = f.readlines()
            if lines:
                last_line = lines[-1].strip()
                # Print progress (overwrite line)
                sys.stdout.write(f"\rStatus: {last_line[:80]}...")
                sys.stdout.flush()

            content = "".join(lines[-20:]) # Check last 20 lines
            
            # Check for success
            if "Application startup complete" in content or "Uvicorn running on" in content:
                print("\n\n✅ ✨ Server Ready! Models loaded (CPU Offload enabled).")
                server_ready = True
                break
            
            # Check for failure
            if "Traceback" in content or "CUDA out of memory" in content:
                print("\n\n❌ Server Failed! Check logs below:")
                print("".join(lines[-30:]))
                break

        if time.time() - start_time > 1200: # 20 mins timeout
            print("\n\n⚠️ Timeout waiting for server.")
            break
            
        time.sleep(2)
        
    except FileNotFoundError:
        pass
    except KeyboardInterrupt:
        print("\nStopped monitoring.")
        break

if server_ready:
    print("\nYou can now run generation cells!")

import requests
import json
from IPython.display import Audio, display

# Define your song parameters
payload = {
    "caption": "female vocals,rap,modern,hip hibhi, Indian fusion,whispered.",
    "duration": 15,    # Seconds
    "thinking": True,  # Use LLM reasoning
    "inference_steps": 25
}

print("🎵 Generating Music...")
try:
    response = requests.post("http://localhost:8000/generate", json=payload)
    
    if response.status_code == 200:
        data = response.json()
        audio_path = data['audios'][0]['path']
        print(f"✅ Success! File saved at: {audio_path}")
        
        # Play the audio directly in Colab
        display(Audio(audio_path))
    else:
        print(f"❌ Error {response.status_code}: {response.text}")

except Exception as e:
    print(f"🚨 Request failed: {e}")
